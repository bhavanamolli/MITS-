import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

# Load dataset
data = pd.read_csv('internship_applications.csv')

# Data preprocessing
data['application_date'] = pd.to_datetime(data['application_date'])
data.dropna(subset=['internship_domain'], inplace=True)

# Applications count per domain
domain_counts = data['internship_domain'].value_counts()

# Plotting
plt.figure(figsize=(10,6))
sns.barplot(x=domain_counts.index, y=domain_counts.values, palette='viridis')
plt.title('Number of Internship Applications per Domain')
plt.xlabel('Internship Domain')
plt.ylabel('Number of Applications')
plt.xticks(rotation=45)
plt.tight_layout()
plt.show()

# Trend over time (monthly)
data['month'] = data['application_date'].dt.to_period('M')
monthly_trends = data.groupby(['month', 'internship_domain']).size().unstack(fill_value=0)

monthly_trends.plot(figsize=(12,7))
plt.title('Monthly Application Trends by Internship Domain')
plt.xlabel('Month')
plt.ylabel('Number of Applications')
plt.legend(title='Domain', bbox_to_anchor=(1.05, 1), loc='upper left')
plt.tight_layout()
plt.show()
